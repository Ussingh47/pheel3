'use client';

import { motion, useScroll, useTransform } from 'framer-motion';
import { useRef, useState, useEffect } from 'react';
import { Volume2, VolumeX } from 'lucide-react';

export default function Hero() {
    const containerRef = useRef<HTMLDivElement>(null);
    const videoRef = useRef<HTMLVideoElement>(null);
    const [isMuted, setIsMuted] = useState(true);
    const [videoError, setVideoError] = useState(false);

    const { scrollYProgress } = useScroll({
        target: containerRef,
        offset: ['start start', 'end start'],
    });

    const y = useTransform(scrollYProgress, [0, 1], ['0%', '50%']);
    const opacity = useTransform(scrollYProgress, [0, 0.5], [1, 0]);

    // set a reasonable default volume once on mount, do not override on mute toggles
    useEffect(() => {
        if (videoRef.current) {
            videoRef.current.volume = 0.5;
        }
    }, []);

    useEffect(() => {
        if (videoRef.current) {
            videoRef.current.muted = isMuted;
        }
    }, [isMuted]);

    const toggleMute = () => {
        setIsMuted(!isMuted);
    };

    const handleVideoError = () => {
        console.error('Background video failed to load or play');
        setVideoError(true);
    };

    return (
        <section id="home" ref={containerRef} className="relative h-screen flex items-center justify-center overflow-hidden bg-black border-b-0">
            {/* Background Video/Animation */}
            <motion.div
                style={{ y, opacity }}
                className="absolute inset-0 z-0"
            >
                <div className="absolute inset-0 bg-black/30 z-10" />
                <video
                    ref={videoRef}
                    autoPlay
                    loop
                    muted
                    playsInline
                    onError={handleVideoError}
                    aria-hidden="true"
                    tabIndex={-1}
                    className="w-full h-full object-cover"
                >
                    <source src="/assets/pheel_vid.mp4" type="video/mp4" />
                    <source src="/assets/pheel_vid.webm" type="video/webm" />
                    <source src="/assets/pheel_vid.ogg" type="video/ogg" />
                </video>
                {videoError && (
                    <div className="absolute inset-0 bg-black flex items-center justify-center">
                        {/* fallback poster image */}
                        <img
                            src="/assets/pheel_vid_poster.jpg"
                            alt="Pheel background"
                            className="object-cover w-full h-full"
                        />
                    </div>
                )}
            </motion.div>

            {/* Content */}
            {/* <div className="relative z-10 text-center px-4 mix-blend-difference">
                <motion.h1
                    initial={{ opacity: 0, y: 100 }}
                    animate={{ opacity: 1, y: 0 }}
                    transition={{ duration: 1, ease: [0.76, 0, 0.24, 1], delay: 0.2 }}
                    className="text-8xl md:text-[12rem] leading-none font-bold tracking-tighter text-white"
                >
                    PHEEL
                </motion.h1>
                <motion.div
                    initial={{ opacity: 0, y: 100 }}
                    animate={{ opacity: 1, y: 0 }}
                    transition={{ duration: 1, ease: [0.76, 0, 0.24, 1], delay: 0.4 }}
                >
                    <span className="text-4xl md:text-7xl font-light tracking-widest text-cyan-500 block mt-[-10px] md:mt-[-20px]">
                        THE MOMENT
                    </span>
                </motion.div>
            </div> */}



            {/* Mute Toggle Button */}
            <motion.button
                initial={{ opacity: 0 }}
                animate={{ opacity: 1 }}
                transition={{ delay: 1, duration: 1 }}
                onClick={toggleMute}
                aria-label={isMuted ? 'Unmute' : 'Mute'}
                aria-pressed={!isMuted}
                className="absolute bottom-8 left-8 z-30 p-4 rounded-full bg-white/5 backdrop-blur-sm border border-white/10 hover:bg-white/10 transition-colors group"
            >
                {isMuted ? (
                    <VolumeX size={24} className="text-white/70 group-hover:text-white transition-colors" />
                ) : (
                    <Volume2 size={24} className="text-white/70 group-hover:text-white transition-colors" />
                )}
            </motion.button>
        </section>
    );
}
